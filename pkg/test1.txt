 CREATE TABLE "word_in_paragraphs" (
	`id` integer PRIMARY KEY AUTOINCREMENT,
	`text` text NOT NULL,
	`sw` text,
	`phonetic` text,
	`definition` text,
	`translation` text,
	`pos` text,
	`collins` integer DEFAULT 0,
	`oxford` integer DEFAULT 0,
	`tag` text,
	`bnc` integer,
	`frq` integer,
	`exchange` text,
	`start` integer,
	`end` integer,
	`subtitle_id` integer,
	`paragraph_id` integer,
	`profile_id` integer,
	CONSTRAINT `fk_word_in_paragraphs_profile` FOREIGN KEY (`profile_id`) REFERENCES `words`(`id`),
	CONSTRAINT `fk_word_in_paragraphs_subtitle` FOREIGN KEY (`subtitle_id`) REFERENCES `subtitles`(`id`),
	CONSTRAINT `fk_word_in_paragraphs_paragraph` FOREIGN KEY (`paragraph_id`) REFERENCES `paragraphs`(`id`),
	CONSTRAINT `fk_paragraphs_words` FOREIGN KEY (`paragraph_id`) REFERENCES `paragraphs`(`id`),
	CONSTRAINT `fk_subtitles_words` FOREIGN KEY (`subtitle_id`) REFERENCES `subtitles`(`id`)
)
