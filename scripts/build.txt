# 打包 windows
go build -a -ldflags="-s -w" -o sqliteweb

# 打包 linux

set GOOS=linux
set GOARCH=amd64
set CGO_ENABLED=0
go build -a -ldflags="-s -w" -o sqliteweb

docker run --rm -v ${PWD}:/output sqliteweb cp /app/sqliteweb /output/
upx .\sqliteweb
